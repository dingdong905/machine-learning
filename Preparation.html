def file2matrix(filename):
    """
    
    parameters:
        fileName - 
    return:
        featureMat - 
        classLabelVector - (didntLike - 0, smallDoses - 1, largeDoses - 2)
    """
    file = open(filename, 'r')  #  
    featureMat = []
    classLabelVector = []
    for line in file.readlines():
        line = line.strip().split('\t')  # 
        features = []
        for feature in line:
            if feature == 'didntLike':  # 
                classLabelVector.append(0)
            elif feature == 'smallDoses':
                classLabelVector.append(1)
            elif feature == 'largeDoses':
                classLabelVector.append(2)
            else:
                features.append(float(feature))  # 
        featureMat.append(features)
    return featureMat, classLabelVector

def showdatas(datingDataMat, datingLabels):
    """
    
    parameters:
        datingDataMat - 
        datingLabels - (didntLike - 0, smallDoses - 1, largeDoses - 2)
    """
    red = []  # 
    yellow = [] # 
    black = [] # 
    for idx in range(len(datingLabels)): # 
        if datingLabels[idx] == 0:
            black.append(datingDataMat[idx])
        elif datingLabels[idx] == 1:
            yellow.append(datingDataMat[idx])
        else:
            red.append(datingDataMat[idx])
    black = np.array(black) # numpy
    yellow = np.array(yellow)
    red = np.array(red)
    plt.figure(figsize=(15, 15)) # 
    plt.subplot(3, 1, 1) # 
    plt.scatter(black[:, 0], black[:, 1], label='didntLike', c='k') #   c: label:
    plt.scatter(yellow[:, 0], yellow[:, 1], label='smallDoses', c='y')
    plt.scatter(red[:, 0], red[:, 1], label='largeDoses', c='r')
    plt.xlabel('Number of frequent flyer miles earned per year') # x
    plt.ylabel('Percentage of time spent playing video games') # y
    plt.legend() # 
    plt.subplot(3, 1, 2)
    plt.scatter(black[:, 0], black[:, 2], label='didntLike', c='k')
    plt.scatter(yellow[:, 0], yellow[:, 2], label='smallDoses', c='y')
    plt.scatter(red[:, 0], red[:, 2], label='largeDoses', c='r')
    plt.xlabel('Number of frequent flyer miles earned per year')
    plt.ylabel('Liters of ice cream consumed per week')
    plt.legend()
    plt.subplot(3, 1, 3)
    plt.scatter(black[:, 1], black[:, 2], label='didntLike', c='k')
    plt.scatter(yellow[:, 1], yellow[:, 2], label='smallDoses', c='y')
    plt.scatter(red[:, 1], red[:, 2], label='largeDoses', c='r')
    plt.ylabel('Percentage of time spent playing video games')
    plt.ylabel('Liters of ice cream consumed per week')
    plt.legend()
    plt.show() #
