def calcShannonEnt(dataSet):
    """
    :
    parameters:
        dataSet - 
    returns:
        shannonEnt - 
    """
    numEntries = len(dataSet)  # 
    labelCounts = {}  # 
    # 
    for featureVec in dataSet:
        label = featureVec[-1]  # 
        if label not in labelCounts.keys():
            labelCounts[label] = 1
        else:
            labelCounts[label] += 1
    # 
    shannonEnt = 0.0
    for key in labelCounts.keys():
        val = float(labelCounts[key]) / numEntries
        shannonEnt -= val * math.log(val, 2)
    return shannonEnt

def splitDataSet(dataSet, axis, value):
    """
    : 
    parameters:
        dataSet - 
        axis -  (axis)
        value - 
    returns:
        retDataSet - 
    """
    reDataSet = []
    for featureVector in dataSet:
        if featureVector[axis] == value:
            reduceFeature = featureVector[:axis]
            reduceFeature.extend(featureVector[axis + 1:])
            reDataSet.append(reduceFeature)
    return reDataSet

def chooseBestFeatureToSplit(dataSet):
    """
    : 
    parameters:
        dataSet - 
    returns:
        bestFeature - ()
    """
    numOfFeature = len(dataSet[0]) - 1
    baseShannon = calcShannonEnt(dataSet)  #
    bestShannon = 0.0
    bestFeature = -1
    for i in range(numOfFeature):
        featureList = [featureVector[i] for featureVector in dataSet]  # i
        featureSet = set(featureList)  # 
        newShannon = 0.0
        for value in featureSet:
            subDataSet = splitDataSet(dataSet, i, value)  # 
            # 
            prob = float(len(subDataSet)) / float(len(dataSet))
            newShannon += prob * calcShannonEnt(subDataSet)
        shannon = baseShannon - newShannon
        if shannon > bestShannon:
            bestShannon = shannon
            bestFeature = i
    return bestFeature

    """
    :

    classList = [example[-1] for example in dataSet]

def classify(inputTree, featLabels, testVec):
    """
    :
    Parameters:
        inputTree - 
        featLabels - 
        testVec - 
    Returns:
        classLabel - 
    """
    firstStr = list(inputTree.keys())[0]
    secondDict = inputTree[firstStr]
    featIndex = featLabels.index(firstStr)
    for key in secondDict.keys():
        if testVec[featIndex] == key:
            if type(secondDict[key]).__name__ == 'dict':
                classLabel = classify(secondDict[key], featLabels, testVec)
            else:
                classLabel = secondDict[key]
    return classLabel

def getTreeDepth(myTree):
    """
        
            
            y
        Parameters
            myTree
        Return:
            maxDepth
    """
    maxDepth = 0
    # firstStr = myTree.keys()[0]
    firstStr = next(iter(myTree))   # key
    secondDict = myTree[firstStr]
    for key in secondDict.keys():
        if type(secondDict[key]).__name__ == 'dict':
            #  
            thisDepth = 1 + getTreeDepth(secondDict[key])
        else:
            #    
            #   if
            thisDepth = 1
        if thisDepth > maxDepth:
            maxDepth = thisDepth
    return maxDepth
